Retrieve the last N elements of a an array-
Write a function that retrieves the last n elements from an array.
```js
// Examples
last([1, 2, 3, 4, 5], 1) ➞ [5]
last([4, 3, 9, 9, 7, 6], 3) ➞ [9, 7, 6]
last([1, 2, 3, 4, 5], 7) ➞ "invalid"
last([1, 2, 3, 4, 5], 0) ➞ []

// My code
function last(a, n) {
	if (n === 0) {
		return []
	}
	if (n > a.length) {
		return "invalid"
	}
	return a.slice(-n)
}
```
